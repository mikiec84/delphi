   double log_normpdf(double x, double mean, double sd) {
     double var = pow(sd, 2);
     double log_denom = -0.5 * log(2 * M_PI) - log(sd);
     double log_nume = pow(x - mean, 2) / (2 * var);
   
     return log_denom - log_nume;
   }
